107        /**
108         * Pregunta el nº de lanzamientos a realizar y los realiza.
109         * 
110         * @param crupier Crupier que lanza su ruleta.
111         */
112        private static void preguntarNumeroLanzamientos(Crupier crupier) {
113                PreguntasLanzamientos lanzamiento = new PreguntasLanzamientos();
114
115                int minLanzamientos = 1;
116                int maxLanzamientos = 1000000;
117                int lanzamientos;
118                String mensajeJugadoresRetirados;
119
120                lanzamientos = lanzamiento.numLanzamientos(minLanzamientos, maxLanzamientos);
121
122                mensajeJugadoresRetirados = crupier.lanzar(lanzamientos);
123
124                lanzamiento.mostrar(mensajeJugadoresRetirados);
125
126                if (lanzamiento.lazarDeNuevo())
127                        preguntarNumeroLanzamientos(crupier);
128
129        }
