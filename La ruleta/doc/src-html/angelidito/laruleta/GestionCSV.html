036        /**
037         * Crea o sobre escribe un CSV con el nombre dado y el contenido de la lista. La
038         * primera fila contendrá el tamaño de las lista.
039         * 
040         * @param <E>           Clase contenida en la lista.
041         * 
042         * @param nombreFichero Nombre completo del fichero, inluidos el punto y la
043         *                      extensión. Ejemplo: nombre.ext
044         * @param lista         Contenido que tendrá el CSV. Cada elemento será una
045         *                      columna.
046         */
047        public static <E> void escribirCSV(String nombreFichero, List<E> lista) {
048
049                File fichero = new File(nombreFichero);
050                fichero.delete();
051
052                FileWriter fileWriter = null;
053                PrintWriter printWriter = null;
054                try {
055                        fileWriter = new FileWriter(nombreFichero, true);
056                        printWriter = new PrintWriter(fileWriter);
057
058                        System.out.printf("\t->Escribiendo en el fichero %s%n", nombreFichero);
059
060                        // Con esto se sobre escribe el fichero
061//                      System.out.printf("Escribiendo linea: %d%n", lista.size());
062
063                        printWriter.println(String.format("%d", lista.size()));
064
065                        for (E elem : lista) {
066//                              System.out.printf("Escribiendo linea: %s%n", elem);
067                                printWriter.println(elem);
068                        }
069
070                        System.out.println("   ->Tarea terminada.\n");
071                } catch (IOException e) {
072                        System.err.println("ERROR.\n" + GestionCSV.class);
073                        e.printStackTrace();
074                } finally {
075                        try {
076                                if (null != fileWriter)
077                                        fileWriter.close();
078                        } catch (Exception e2) {
079                                e2.printStackTrace();
080                        }
081                }
082        }

135        /**
136         * Recupera y devuelve el histórico a partir del fichero historico.csv. Si no
137         * tiene el formato deseado lo inicializa a 0. Si algun valor es negativo se
138         * toma como 0. Si no existe historico.csv, lo crea con los datos a 0.
139         * 
140         * @return El histórico a partir de historico.csv
141         */
142        public static Integer[] obtenerHistoricoOcurrencias() {
143
144                Integer[] ocurrencias = new Integer[37];
145
146                String nombreFichero = "historico.csv";
147
148                File fichero = new File(nombreFichero);
149
150                FileReader fileReader = null;
151                BufferedReader bufferedReader = null;
152
153                try {
154
155                        if (!fichero.exists())
156                                throw new NoCSVException(String.format(
157                                                "No se ha encontrado el fichero %s"
158                                                                + "%nPor favor, no se preocupe si es la primera vez que inicia el programa.",
159                                                nombreFichero));
160
161                        fileReader = new FileReader(fichero);
162                        bufferedReader = new BufferedReader(fileReader);
163
164                        String linea = bufferedReader.readLine();
165//                      System.out.println(linea);
166                        int lineasLeidas = 0;
167                        if (linea != null && Integer.parseInt(linea) == ocurrencias.length)
168                                while (lineasLeidas < ocurrencias.length) {
169                                        linea = bufferedReader.readLine();
170                                        // Este método puede lanzar NumberFormatException, que se maneja en un catch
171                                        int cantidad = Integer.parseInt(linea);
172
173                                        if (cantidad > 0)
174                                                ocurrencias[lineasLeidas] = cantidad;
175                                        else
176                                                ocurrencias[lineasLeidas] = 0;
177
178//                                      System.out.println(lineasLeidas + "  -  " + ocurrencias[lineasLeidas]);
179
180                                        lineasLeidas++;
181                                }
182
183                        if (lineasLeidas < ocurrencias.length)
184                                throw new CantidadCamposCSVException("Cantidad de filas no adecuada en el fichero historico.csv");
185
186                } catch (NoCSVException e) {
187
188                        System.err.println("No se ha podido cargar el historial de la Ruleta.");
189                        System.err.println(e.getMessage());
190                        System.err.println();
191
192                        for (int i = 0; i < ocurrencias.length; i++)
193                                ocurrencias[i] = 0;
194
195                        GestionCSV.escribirCSV(nombreFichero, Arrays.asList(ocurrencias));
196
197                } catch (BadCSVException e) {
198
199                        System.err.println("No se ha podido cargar el historial de la Ruleta.");
200                        System.err.println(e.getMessage());
201                        System.err.println();
202
203                        for (int i = 0; i < ocurrencias.length; i++)
204                                ocurrencias[i] = 0;
205
206                } catch (NumberFormatException e) {
207
208                        System.err.println("No se ha podido cargar el historial de la Ruleta.");
209                        System.err.println("Contenido no numérico en " + nombreFichero + ", se procede a formatear su contenido.");
210
211                        System.err.println();
212                        for (int i = 0; i < ocurrencias.length; i++)
213                                ocurrencias[i] = 0;
214
215                        GestionCSV.escribirCSV(nombreFichero, Arrays.asList(ocurrencias));
216
217                } catch (IOException e) {
218
219                        System.err.println("ERROR INESPERADO");
220                        System.err.println("No se ha podido cargar el historial de la Ruleta.");
221
222                        System.err.println();
223                        e.printStackTrace();
224
225                        for (int i = 0; i < ocurrencias.length; i++)
226                                ocurrencias[i] = 0;
227
228                } finally {
229
230                        try {
231
232                                if (fileReader != null)
233                                        fileReader.close();
234
235                        } catch (IOException e) {
236
237                                System.err.println("ERROR INESPERADO");
238
239                                System.err.println();
240                                e.printStackTrace();
241
242                        }
243                }
244//              for (Integer integer : ocurrencias) {
245//
246//                      System.out.println(integer);
247//              }
248
249                return ocurrencias;
250        }

252        /**
253         * Recupera y devuelve la lista jugadores a partir de un fichero. Si alguna
254         * linea no contiene los datos necesarios para crear un jugador, no lo añade a
255         * la lista. Informa de cualquier error que pueda suceder.
256         * 
257         * @param nombreFichero nombre del fichero que almacena los jugadores.
258         * @return ArrayList con los jugadores que contenia el fichero.
259         */
260        public static ArrayList<Jugador> obtenerJugadores(String nombreFichero) {
261
262                ArrayList<Jugador> jugadores = new ArrayList<Jugador>();
263
264                FileReader fr = null;
265                BufferedReader br = null;
266
267                try {
268
269                        try {
270                                // Esto lanza FileNotFoundException
271                                // que es subclase de IOException
272                                fr = new FileReader(nombreFichero);
273
274                        } catch (IOException e) {
275
276                                throw new NoCSVException(String.format(
277                                                "No se ha encontrado el fichero %s"
278                                                                + "%nPor favor, no se preocupe si es la primera vez que inicia el programa.",
279                                                nombreFichero));
280                                // Respecto a lo de arriba->tmpco se precupe si hay + de 1 objeto ListaJugadores
281                        }
282                        br = new BufferedReader(fr);
283
284                        String lineaCSV;
285                        int lineasQueLeer;
286
287                        try {
288
289                                lineaCSV = br.readLine();
290
291                                if (lineaCSV == null)
292                                        throw new ContenidoNullCSVException(String.format("Informacion irrecuperable en el fichero %s"
293                                                        + "%nEn la primera fila y columna deberia estar la cantidad de jugadores "
294                                                        + "que guarda el fichero, sin embargo el contenido es nulo.", nombreFichero));
295
296                                // La primera linea del CSV contiene la cantidad de filas del CSV a leer
297                                lineasQueLeer = Integer.parseInt(lineaCSV);
298
299                        } catch (NumberFormatException e) {
300                                throw new ContenidoCorruptoCSVException(String.format("Informacion irrecuperable en el fichero %s"
301                                                + "%nEn la primera fila y columna deberia estar la cantidad de jugadores que guarda el fichero."
302                                                + "%nSe recomienda correción manual.", nombreFichero));
303                        }
304
305                        for (int i = 0; i < lineasQueLeer; i++) {
306
307                                lineaCSV = br.readLine();
308
309                                try {
310                                        jugadores.add(new Jugador(lineaCSV));
311                                } catch (BadCSVException e) {
312//                                      System.err.println(e.getMessage());
313                                        System.err.println();
314                                }
315
316                        } // Fin del for
317
318                } catch (BadCSVException e) {
319
320                        System.err.println();
321//                      System.err.println(e.getMessage());
322//                      e.printStackTrace();
323
324                } catch (IOException e) {
325                        e.printStackTrace();
326                }
327
328                finally {
329                        try {
330                                if (fr != null)
331                                        fr.close();
332                        } catch (IOException e) {
333                                e.printStackTrace();
334                        }
335
336                }
337
338                return jugadores;
339        }

344        public static void borrarDatos() {
345

347
348                String ficheroHistorico = "historico.csv";

356                        eliminarFichero(fichero);
357
358                }
359
360                String ficheroRetirados = "jugadoresRetirados.csv";
361                fichero = new File(ficheroRetirados);
362                eliminarFichero(fichero);
363
364        }
