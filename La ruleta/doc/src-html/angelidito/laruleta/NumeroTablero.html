006/**
007 * 
008 * 
009 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
010 *
011 */
012public class NumeroTablero extends Number implements Comparable<NumeroTablero> {
013
014        /**
015         * 
016         */
017        private static final long serialVersionUID = -3233488705262267331L;
018
019        /**
020         * Numero y valor al que se corresponde el objeto.
021         */
022        private final int n;
023        /**
024         * 0 si par o cero, 1 si impar
025         */
026        private int paridad;
027        /**
028         * -1 si cero, 0 si rojo, 1 si negro
029         */
030        private int color;
031        /**
032         * -1, 0 o 1, para el cero, la primera o segunda mitad, respectivamente.
033         */
034        private int mitad;
035        /**
036         * -1, 0, 1 o 2, para el cero, la primera, segunda o tercera docena,
037         * respectivamente.
038         */
039        private int docena;
040        /**
041         * -1, 0, 1 o 2, para el cero, la fila del 1, del 2 y del 3, respectivamente.
042         */
043        private int fila;
044
045        /**
046         * Encaja los atributos de la clase en los valores que les corresponden en un
047         * tablero de ruleta francesa. Si el número no existe en el tablero o es 0, los
048         * valores de todos los atributos salvo de {@code n} seran iguales -1.
049         * 
050         * @param numero El numero al que corresponde el objeto.
051         */
052        public NumeroTablero(int numero) {
053
054                this.n = numero;
055
056                // Para 0 o los numeros que no pertenecen al tablero
057                if (n < 1 || 36 < n) {
058                        paridad = -1;
059                        color = -1;
060                        mitad = -1;
061                        docena = -1;
062                        fila = -1;
063                }
064
065                // Para los numeros del tablero sin contara el 0
066                else {
067                        paridad = n % 2;
068
069                        if ((0 < n && n < 10 && n % 2 == 1) || (11 < n && n < 19 && n % 2 == 0) || (18 < n && n < 28 && n % 2 == 1)
070                                        || (29 < n && n < 37 && n % 2 == 0))
071                                color = 0;
072                        else
073                                color = 1;
074
075                        if (n < 19)
076                                mitad = 0;
077                        else
078                                mitad = 1;
079
080                        if (n < 13)
081                                docena = 0;
082                        else if (n < 25)
083                                docena = 1;
084                        else
085                                docena = 2;
086
087                        if (n % 3 == 1)
088                                fila = 1;
089                        else if (n % 3 == 2)
090                                fila = 2;
091                        else
092                                fila = 0;
093                }
094        }
095
096        /**
097         * @return El número al que corresponde.
098         */
099        public int getN() {
100                return n;
101        }
102
103        /**
104         * @return 0 si par, 1 si impar; -1 si cero
105         */
106        public int getParidad() {
107                return paridad;
108        }
109
110        /**
111         * @return 0 si rojo, 1 si negro; -1 si cero.
112         */
113        public int getColor() {
114                return color;
115        }
116
117        /**
118         * @return 0 o 1, para la primera o segunda mitad, respectivamente; -1 si cero.
119         */
120        public int getMitad() {
121                return mitad;
122        }
123
124        /**
125         * @return 0, 1 o 2, para la primera, segunda o tercera docena,
126         *         respectivamente; -1 si cero.
127         */
128        public int getDocena() {
129                return docena;
130        }
131
132        /**
133         * @return 0, 1 o 2, para la fila del 3, del 1 y del 2,
134         *         respectivamente; -1 si cero.
135         */
136        public int getFila() {
137                return fila;
138        }
139
140        /**
141         * Compara este Numero con el especificado en el parámetro. Devuelve un entero
142         * negativo, cero, o un entero positivo si el Numero es menor, igual o mayor que
143         * el Numero del parámetro.
144         * 
145         * @param numero El Numero a comparar.
146         * @return un entero negativo, cero, o un entero positivo si el objeto es
147         *          menor, igual o mayor que el objeto del parámetro.
148         */
149        @Override
150        public int compareTo(NumeroTablero numero) {
151
152                int diferencia;
153
154                diferencia = this.n - numero.getN();
155
156                return diferencia;
157        }
158
159        @Override
160        public String toString() {
161                return String.format("%d", n);
162        }
163
164        @Override
165        public int intValue() {
166                return n;
167        }
168
169        @Override
170        public long longValue() {
171                return (long) n;
172        }
173
174        @Override
175        public float floatValue() {
176                return (float) n;
177        }
178
179        @Override
180        public double doubleValue() {
181                return (double) n;
182        }
183
184}
