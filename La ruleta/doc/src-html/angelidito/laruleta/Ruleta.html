018        private int ultimoNumero;
019
020        /**

022         * a 36, el valor de la posición indica el número de ocurrencias del número

039        private static Random random;

041        static {
042                random = new Random();
043                historico = new NumeroRuleta[37];
044                recuperarHistorico();
045                topTen = new NumeroRuleta[10];

052        /**
053         * Recupera el histórico.
054         * 
055         */
056        private static void recuperarHistorico() {
057                setHistorico(GestionCSV.obtenerHistoricoOcurrencias());
058        }
059
060        /**
061         * Fija las ocurrencias del histórico a partir del vector dado. Si no tiene el
062         * tamaño deseado las inicializa a 0. Si algun valor es negativo se toma como 0.
063         * 
064         * @param ocurrencias Vector con ocurrencias de cada número.
065         */
066        private static void setHistorico(Integer[] ocurrencias) {
067                if (Ruleta.historico.length != ocurrencias.length)
068                        for (int i = 0; i < Ruleta.historico.length; i++)
069                                Ruleta.historico[i] = new NumeroRuleta(i);
070                else
071                        for (int i = 0; i < Ruleta.historico.length; i++)
072                                if (ocurrencias[i] < 0)
073                                        Ruleta.historico[i] = new NumeroRuleta(i, ocurrencias[i]);
074                                else
075                                        Ruleta.historico[i] = new NumeroRuleta(i);
076
077        }
078
079        // tirar seguidas? quizá eso sea mejor en Crupier Y AHORRAMOS LIOS
080        /**
081         * Genera un número aleatório del 0 al 36. Lo devuelve y lo guarda. También lo
082         * anota en el histórico.
083         * 
084         * @return Un número al alzar entre 0 y 36, incluidos.
085         */
086        public int lanzar() {
087
088                this.ultimoNumero = Ruleta.random.nextInt(37);
089
090                Ruleta.historico[ultimoNumero].añadirOcurrencia();
091                ++Ruleta.totalTiradas;
092
093                return ultimoNumero;
094        }
095
096        public static String estadisticas() {
097
098                String estadisticas = "";
099                for (NumeroRuleta numero : historico) {
100                        estadisticas += String.format("Nº%d: %d ocurrencias; %.3f", numero.getN(), numero.getOcurrencias(),
101                                        (float) numero.getOcurrencias() * 100 / Ruleta.totalTiradas)+"%\n";
102                }
103
104                return estadisticas;
105        }
106
107        /**
108         * Devuelve el historial de ocurrencias.
109         * 
110         * @return El histórico
111         */
112        public static NumeroRuleta[] getHistorico() {
113                return historico;
114        }
115
116        public NumeroRuleta getNumeroRuleta(int n) {
117
118                return Ruleta.historico[n];
119
120        }
121
122        /**
123         * Devuelve el total de tiradas.
124         * 
125         * @return El total de tiradas.
126         */
127        public static int getTotalTiradas() {
128                return totalTiradas;
129        }
130
131        /**
132         * Guarda el historico en un CSV. Guarda en orden las ocurrencias de cada nº. La
133         * primera fila será 37 (el nº de filas que vendrán despues), las siguientes
134         * contendrán las ocurrencias.
135         */
136        public static void guardarHisorico() {
137
138                String nombreFichero = "historico.csv";
139
140                Integer[] ocurrencias = new Integer[Ruleta.historico.length];
141                for (int i = 0; i < ocurrencias.length; i++) {
142                        ocurrencias[i] = Ruleta.historico[i].getOcurrencias();
143                }
144
145                List<Integer> listaHistorico = Arrays.asList(ocurrencias);
146
147                GestionCSV.escribirCSV(nombreFichero, listaHistorico);
148
149        }
150
151        /**
152         * Crea la Ruleta y restaura el historial de números. Restaura el histórico si
153         * existe el fichero que guarda sus datos.
154         */
155        public Ruleta() {
156
157                this.ultimoNumero = 0;
158
159        }
