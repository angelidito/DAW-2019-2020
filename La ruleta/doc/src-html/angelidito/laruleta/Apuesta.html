124        /**
125         * Recupera la Apuesta del CSV. Crea el objeto con los vectores que sacará del
126         * parámetro. Cuando alguno estos no tenga la dimensión acertada, fijará a cero
127         * esa parte de la apuesta. Si en algún vector alguna cantidad a apostar es
128         * negativa, esta se interpretará como 0. Si al dividir el parámetro por el
129         * signo almohadilla, "#", el tamaño del vector resultante no es de longitud 6,
130         * la apuesta será iniciada como el constructor por defecto: a cero. Aumenta el
131         * contador numApuestas.
132         * 
133         * @param ultimaColumnaDelCSV La úlima columna del CSV que debería contener el
134         *                            texto creado por el método toFormatoCSV().
135         * @throws CantidadCamposCSVException Cuando la cantidad de campos
136         */
137        public Apuesta(String ultimaColumnaDelCSV) throws BadCSVException {
138
139                // Esto o deberia pasar por cómo está programado el método que lee las lineas
140                // de los ficheros, pero por si acaso
141                if (ultimaColumnaDelCSV == null)
142                        throw new ContenidoNullCSVException("Contenido nulo al recuperar la acpuesta:");
143
144                String[] vectores = ultimaColumnaDelCSV.split("#");
145                // Si coincide la longitud, bien; si no, todo a 0.
146                if (vectores.length != 6)
147                        throw new CantidadCamposCSVException("No hay campos suficientes para recuperar la puesta:");
148
149                // Vectores contiene:
150                // numeros, parImpar, rojoNegro,
151                // mitades, decenas y modulos; en ese orden,
152                String[] cantidadesANumerosStr = vectores[0].split(":");
153                String[] parImparStr = vectores[1].split(":");
154                String[] rojoNegroStr = vectores[2].split(":");
155                String[] mitadesStr = vectores[3].split(":");
156                String[] docenasStr = vectores[4].split(":");
157                String[] modulosStr = vectores[5].split(":");
158
159                // Numeros
160                // Si coincide la longitud, bien; si no, todo a 0.
161                // Esta estructura se repite en los siguientes if-else
162                if (this.numeros.length != cantidadesANumerosStr.length)
163                        this.setNumerosTo0();
164                else
165                        this.setNumeros(transformarVector(cantidadesANumerosStr));
166
167                // ParImpar
168                if (this.parImpar.length != parImparStr.length)
169                        this.setParImparTo0();
170                else
171                        this.setParImpar(transformarVector(parImparStr));
172
173                // RojoNegro
174                if (this.rojoNegro.length != rojoNegroStr.length)
175                        this.setRojoNegroTo0();
176                else
177                        this.setRojoNegro(transformarVector(rojoNegroStr));
178
179                // Mitades
180                if (this.mitades.length != mitadesStr.length)
181                        this.setMitadesTo0();
182                else
183                        this.setMitades(transformarVector(mitadesStr));
184
185                // Docenas
186                if (this.docenas.length != docenasStr.length)
187                        this.setDocenasTo0();
188                else
189                        this.setDocenas(transformarVector(docenasStr));
190
191                // Modulos
192                if (this.modulos.length != modulosStr.length)
193                        this.setModulosTo0();
194                else
195                        this.setModulos(transformarVector(modulosStr));
196
197                numApuestas++;
198
199        }

205        /**
206         * Transforma un vector de tipo String a uno de enteros. El contenido del vector
207         * debe de ser analizable como números enteros, en caso de no ser así, devuelve
208         * un vector del mismo tamaño inicializado a 0.
209         * 
210         * @param vectorDeNumeros Vector de numeros en formato String.
211         * @return Un vector de enteros.
212         */
213        private int[] transformarVector(String[] vectorDeNumeros) {
214
215                int[] vectorRecuperado = new int[vectorDeNumeros.length];
216
217                try {
218                        for (int i = 0; i < vectorDeNumeros.length; i++)
219                                vectorRecuperado[i] = Integer.parseInt(vectorDeNumeros[i]);
220                } catch (NumberFormatException e) {
221                        vectorRecuperado = new int[vectorDeNumeros.length];
222                }
223
224                return vectorRecuperado;
225        }

254        /**
255         * Devuelve la apuesta en formato para CSV. Para poder recuperarla al iniciar de
256         * nuevo el programa. En el CSV irá en una misma celda. Guarda los vectores
257         * numeros, parImpar, rojoNegro, mitades, decenas y modulos; en ese orden,
258         * separados por el singo de almohadilla, "#". Las posiciones de cada vector se
259         * separarán por el signo de dos puntos, ":".
260         * 
261         * @return La apuesta en texto con formato CSV.
262         */
263        @Override
264        public String toString() {
265
266                String enFormato = "";
267
268                enFormato += numeros[0].getCantidadApostada();
269                for (int i = 1; i < numeros.length; i++)
270                        enFormato += ":" + numeros[i].getCantidadApostada();
271
272                enFormato += "#";
273
274                enFormato += parImpar[0];
275                for (int i = 1; i < parImpar.length; i++)
276                        enFormato += ":" + parImpar[i];
277
278                enFormato += "#";
279
280                enFormato += rojoNegro[0];
281                for (int i = 1; i < rojoNegro.length; i++)
282                        enFormato += ":" + rojoNegro[i];
283
284                enFormato += "#";
285
286                enFormato += mitades[0];
287                for (int i = 1; i < mitades.length; i++)
288                        enFormato += ":" + mitades[i];
289
290                enFormato += "#";
291
292                enFormato += docenas[0];
293                for (int i = 1; i < docenas.length; i++)
294                        enFormato += ":" + docenas[i];
295
296                enFormato += "#";
297
298                enFormato += modulos[0];
299                for (int i = 1; i < modulos.length; i++)
300                        enFormato += ":" + modulos[i];
301
302                return enFormato;
303        }

305        /**
306         * Devuelve la informacion de lo apostado. Si no hay apostado nada, devuelve in
307         * texto informando de ello.
308         * 
309         * @return La información de lo apostado.
310         */
311        public String info() {
312                String info = "";
313                int totalAnotado = 0;
314
315                for (NumeroApuesta numero : this.numeros)
316                        if (numero.getCantidadApostada() > 0) {
317                                info += String.format("%d al nº%d. ", numero.getCantidadApostada(), numero.getN());
318                                totalAnotado += numero.getCantidadApostada();
319                        }
320
321                if (totalAnotado < this.totalApostado())
322                        if (this.parImpar[0] > 0) {
323                                info += String.format("%d a pares. ", this.parImpar[0]);
324                                totalAnotado += this.parImpar[0];
325                        }
326
327                if (totalAnotado < this.totalApostado())
328                        if (this.parImpar[1] > 0) {
329                                info += String.format("%d a impares. ", this.parImpar[1]);
330                                totalAnotado += this.parImpar[1];
331                        }
332
333                if (totalAnotado < this.totalApostado())
334                        if (this.rojoNegro[0] > 0) {
335                                info += String.format("%d al rojo. ", this.rojoNegro[0]);
336                                totalAnotado += this.rojoNegro[0];
337                        }
338
339                if (totalAnotado < this.totalApostado())
340                        if (this.rojoNegro[1] > 0) {
341                                info += String.format("%d al negro. ", this.rojoNegro[1]);
342                                totalAnotado += this.rojoNegro[1];
343                        }
344
345                if (totalAnotado < this.totalApostado())
346                        if (this.mitades[0] > 0) {
347                                info += String.format("%d a la primera mitad. ", this.mitades[0]);
348                                totalAnotado += this.mitades[0];
349                        }
350
351                if (totalAnotado < this.totalApostado())
352                        if (this.mitades[1] > 0) {
353                                info += String.format("%d a la segunda mitad. ", this.mitades[1]);
354
355                        }
356
357                if (totalAnotado < this.totalApostado())
358                        if (this.docenas[0] > 0) {
359                                info += String.format("%d a la primera docena. ", this.docenas[0]);
360                                totalAnotado += this.docenas[0];
361                        }
362
363                if (totalAnotado < this.totalApostado())
364                        if (this.docenas[1] > 0) {
365                                info += String.format("%d a la segunda docena. ", this.docenas[1]);
366                                totalAnotado += this.docenas[1];
367                        }
368
369                if (totalAnotado < this.totalApostado())
370                        if (this.docenas[2] > 0) {
371                                info += String.format("%d a la tercera docena. ", this.docenas[2]);
372                                totalAnotado += this.docenas[2];
373                        }
374
375                if (totalAnotado < this.totalApostado())
376                        if (this.modulos[1] > 0) {
377                                info += String.format("%d a la primera fila 2 a 1. ", this.modulos[1]);
378                                totalAnotado += this.modulos[1];
379                        }
380
381                if (totalAnotado < this.totalApostado())
382                        if (this.modulos[2] > 0) {
383                                info += String.format("%d a la segunda fila 2 a 1. ", this.modulos[2]);
384                                totalAnotado += this.modulos[2];
385                        }
386
387                if (totalAnotado < this.totalApostado())
388                        if (this.modulos[0] > 0) {
389                                info += String.format("%d a la tercera fila 2 a 1. ", this.modulos[0]);
390                                totalAnotado += this.modulos[0];
391                        }
392
393                if (totalAnotado == 0)
394                        info = "ninguna";
395
396                if (totalAnotado != this.totalApostado())
397                        throw new BadProgramingRTException("ERROR DE CONTEO EN Apuesta.info()");
398
399                return info;
400        }

402        /**
403         * Devuelve la informacion completa de la puesta. Va por arrays. Método más útil
404         * para pruebas y desarrollo qu epara la apliación final. Se recomienda usar el
405         * método {@code info()}.
406         * 
407         * @return Toda la información de la apuesta.
408         */
409        public String informacion() {
410                int[] cantidades = new int[numeros.length];
411
412                for (int i = 0; i < cantidades.length; i++) {
413                        cantidades[i] = numeros[i].getCantidadApostada();
414                }
415
416                String informacion = "***    numeros   = " + Arrays.toString(cantidades) + "\n***    parImpar  = "
417                                + Arrays.toString(parImpar) + "\n***    rojoNegro = " + Arrays.toString(rojoNegro) + "\n***    mitades   = "
418                                + Arrays.toString(mitades) + "\n***    docenas   = " + Arrays.toString(docenas) + "\n***    modulos   = "
419                                + Arrays.toString(modulos);
420                return informacion;
421        }

427         * Devuelve la suma total del crédito apostado.
428         * 
429         * @return El crédito total apostado
430         */
431        public int totalApostado() {
432                int total = 0;
433                for (NumeroApuesta apuesta : numeros)
434                        total += apuesta.getCantidadApostada();
435                for (int apuesta : parImpar)
436                        total += apuesta;
437                for (int apuesta : rojoNegro)
438                        total += apuesta;
439                for (int apuesta : mitades)
440                        total += apuesta;
441                for (int apuesta : docenas)
442                        total += apuesta;
443                for (int apuesta : modulos)
444                        total += apuesta;
445                return total;
446        }

606        public void setNumerosTo0() {
607

609                        if (numeros[i] == null || numeros[i].getN() != i)
610                                numeros[i] = new NumeroApuesta(i);

613                }
614        }

835        /**
836         * Asegura que la cantidad a apostar no sea negativa. Si el valor es negativo,
837         * devuelve 0; en caso contrario no altera el valor.
838         * 
839         * @param cantidad Valor que no se desea que sea menor que 0.
840         * @return El mismo valor del parámetro cantidad si era mayor que 0, en caso
841         *         contrario devuelve 0.
842         */
843        private int asegurarCantidadNoNegativa(int cantidad) {
844                if (cantidad < 0)
845                        cantidad = 0;
846                return cantidad;
847        }

849        /**
850         * Asegura que las cantidades a apostar no sean negativa. Si algun valor del
851         * vector es negativo, lo vuelve 0; en caso contrario no lo altera.
852         * 
853         * 
854         * @param cantidades Vector de valores que no se desea que sean menores que 0.
855         * @return Un vector con que contendrá en cada posicion el valor que tenía si
856         *         este era mayor que 0 y si era menor, contendrá el valor 0.
857         */
858        private int[] asegurarCantidadNoNegativa(int[] cantidades) {
859                for (int cantidad : cantidades)
860                        cantidad = this.asegurarCantidadNoNegativa(cantidad);
861                return cantidades;
862        }
