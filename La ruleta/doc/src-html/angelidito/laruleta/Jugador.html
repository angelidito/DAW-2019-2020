175        /**
176         * Crea un jugador con la infomacion de una línea del CSV. Da valor a todos los
177         * atributos de la clase con dicha información, salvo si el CSV no contiene las
178         * 7 celdas que debería, en cuyo caso haría lo mismo que el contructor por
179         * defecto.
180         * 
181         * @param lineaDelCSV Línea del CSV que contiene toda la informarción de un
182         *                    jugador.
183         * @throws BadCSVException Cuando la linea del CSV es null, la cantidad de
184         *                         campos que contiene es menor que 8 o hay contenido no
185         *                         numérico donde no debería.
186         */
187        public Jugador(String lineaDelCSV) throws BadCSVException {
188
189                if (lineaDelCSV == null)
190                        throw new ContenidoNullCSVException("Información nula de jugador.");
191
192                try {
193
194                        String[] columnas = lineaDelCSV.split(";");
195
196                        // Si coincide la longitud, bien; si no, excepción y constructor por defecto.
197                        if (columnas.length < 8)
198                                throw new CantidadCamposCSVException(
199                                                "No hay campos suficientes para recuperar el jugador:" + lineaDelCSV);
200
201                        this.credito = Integer.parseInt(columnas[1]);
202                        this.creditoObjetivo = Integer.parseInt(columnas[2]) > 10 ? Integer.parseInt(columnas[2]) : 10;
203                        this.rondasMaximas = Integer.parseInt(columnas[3]) > 10 ? Integer.parseInt(columnas[2]) : 10;
204                        this.rondas = Integer.parseInt(columnas[4]);
205                        this.ganado = Integer.parseInt(columnas[5]);
206                        this.perdido = Integer.parseInt(columnas[6]);
207
208                        // Esta parte viene depues de lo anterior podría haber dado error lo anterior y
209                        // no queremos aumentar numJugadores
210                        this.id = numJugadores++;
211                        // Nombre
212                        if (nombre != null && columnas[0].length() >= 11) {
213
214                                if (columnas[0].substring(0, 9).compareTo("SinNombre") == 0)
215                                        this.nombre = "SinNombre" + String.format("%02d", ++jugadoresSinNombre);
216                                else
217                                        this.nombre = columnas[0];
218
219                        } else if (columnas[0] != null)
220                                this.nombre = columnas[0];
221                        else
222                                this.nombre = "SinNombre" + String.format("%02d", ++jugadoresSinNombre);
223
224                        // Y esto viene lo ultimo porque el catch usa el nombre del jugador.
225                        try {
226                                this.apuesta = new Apuesta(columnas[7]);
227                        } catch (CantidadCamposCSVException e) {
228//                              System.err.println(e.getMessage());
229                                System.err.printf("Apuesta de %s fijada a 0.%n", this.nombre);
230                                System.err.println();
231                                this.apuesta = new Apuesta();
232
233                        }
234
235                } catch (NumberFormatException e) {
236                        throw new ContenidoCorruptoCSVException("No se ha podido cargar un Jugador del CSV: " + lineaDelCSV);
237                }
238
239        }

482        /**
483         * Edita el jugador según la opcion introducida por teclado. Opciones: 1,
484         * nombre; 2, crédito; 3, crédito objetivo;, 4, rondas máximas; 5, apuesta; 0,
485         * nada.
486         * 
487         * @param opcion 1, para cambiar el nombre; 2, para variar el crédito; 3, para
488         *               cambiar el crédito objetivo; 4, para cambiar el máximo número
489         *               de rondas; 5, para editar la apuesta; 0, no hace nada; en
490         *               cualquier otro caso, informa que la opción es incorrecta.
491         * 
492         * @return {@code true} si el jugador ha alcanzado el crédito objetivo o las
493         *         rondas máximas, o si no tiene crédito; de otra manera, {@code false}
494         */
495        public boolean editarJugador(int opcion) {
496
497                boolean retirar = false;
498
499                switch (opcion) {
500
501                case 1:
502                        System.out.println("Introduzca el nuevo nombre.");
503
504                        this.setNombre(Escaner.texto());
505
506                        break;
507
508                case 2:
509                        System.out.println("Inserte la cantidad a sumar (restará si es negativa).");
510
511                        this.variarCredito(
512                                        Escaner.entero(-1 * Jugador.MAX_INT_VALUE + this.credito, Jugador.MAX_INT_VALUE - this.credito));
513
514                        if (this.credito < 1) {
515                                retirar = true;
516                                System.err.println("Sin credito. El jugador se retirará tras la edición");
517                        }
518
519                        break;
520
521                case 3:
522                        System.out.println("Inserte el nuevo crédito objetivo.");
523                        System.out.printf("El actual es: %d.%n", this.getCreditoObjetivo());
524
525                        int nuevoCreditoObjetivo = Escaner.entero(0, Jugador.MAX_INT_VALUE);
526                        this.setCreditoObjetivo(nuevoCreditoObjetivo);
527
528                        if (this.creditoObjetivo <= this.credito) {
529                                retirar = true;
530                                System.err.println("Crédito objetivo superado. El jugador se retirará tras la edición");
531                        }
532
533                        break;
534
535                case 4:
536                        System.out.println("Inserte el nuevo máximo de rondas.");
537                        System.out.printf("El actual es: %d.%n", this.getRondasMaximas());
538
539                        int nuevasRondasMaximas = Escaner.entero(1, Jugador.MAX_INT_VALUE);
540                        this.setCreditoObjetivo(nuevasRondasMaximas);
541
542                        if (this.rondasMaximas <= this.rondas) {
543                                retirar = true;
544                                System.err.println("Rondas máximas superadas. El jugador se retirará tras la edición");
545                        }
546
547                        break;
548
549                case 5:
550
551                        menuEditarApuesta();
552
553                        break;
554
555                case 0:
556                        // NO TIENE QUE HACER NADA.
557                        // Solo salir.
558                        break;
559
560                default:
561                        Escaner.avisoOpcionIncorrecta();
562
563                }
564                return retirar;
565        }
