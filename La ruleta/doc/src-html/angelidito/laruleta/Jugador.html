175        /**
176         * Crea el Jugador con el nombre, el crédito, las rondas máximas y la apuesta
177         * pasados. Aunque se pase un crédito o un numero de rondas másximas menor, los
178         * mínimos serán de 10. Fija la apuesta a la pasada por para, si la que recibe
179         * es null, si no, le fija la del parámetro.

181         * @param nombre          Nombre del jugador.
182         * @param credito         El crédito del jugador.
183         * @param creditoObjetivo Crédito con el que el jugador se retirará.
184         * @param rondasMaximas   Las rondas máximas a jugar.
185         * @param apuesta         Apuesta a realizar.
186         */
187        public Jugador(String nombre, int credito, int creditoObjetivo, int rondasMaximas, Apuesta apuesta) {
188                this.id = numJugadores++;
189                this.nombre = nombre != null ? nombre : "SinNombre" + String.format("%02d", id);
190                this.credito = credito;
191                this.creditoObjetivo = creditoObjetivo;
192                this.rondasMaximas = rondasMaximas > 10 ? rondasMaximas : Jugador.MAX_INT_VALUE;
193                this.apuesta = apuesta != null ? apuesta : new Apuesta();

196        /**
197         * Crea un jugador con la infomacion de una línea del CSV. Da valor a todos los
198         * atributos de la clase con dicha información, salvo si el CSV no contiene las
199         * 7 celdas que debería, en cuyo caso haría lo mismo que el contructor por
200         * defecto.
201         * 
202         * @param lineaDelCSV Línea del CSV que contiene toda la informarción de un
203         *                    jugador.
204         * @throws BadCSVException Cuando la linea del CSV es null, la cantidad de
205         *                         campos que contiene es menor que 8 o hay contenido no
206         *                         numérico donde no debería.
207         */
208        public Jugador(String lineaDelCSV) throws BadCSVException {
209
210                if (lineaDelCSV == null)
211                        throw new ContenidoNullCSVException("Información nula de jugador.");
212
213                try {
214
215                        String[] columnas = lineaDelCSV.split(";");
216
217                        // Si coincide la longitud, bien; si no, excepción y constructor por defecto.
218                        if (columnas.length < 8)
219                                throw new CantidadCamposCSVException(
220                                                "No hay campos suficientes para recuperar el jugador:" + lineaDelCSV);
221
222                        this.credito = Integer.parseInt(columnas[1]);
223                        this.creditoObjetivo = Integer.parseInt(columnas[2]) > 10 ? Integer.parseInt(columnas[2]) : 10;
224                        this.rondasMaximas = Integer.parseInt(columnas[3]) > 10 ? Integer.parseInt(columnas[2]) : 10;
225                        this.rondas = Integer.parseInt(columnas[4]);
226                        this.ganado = Integer.parseInt(columnas[5]);
227                        this.perdido = Integer.parseInt(columnas[6]);
228
229                        // Esta parte viene depues de lo anterior podría haber dado error lo anterior y
230                        // no queremos aumentar numJugadores
231                        this.id = numJugadores++;
232                        // Nombre
233                        if (nombre != null && columnas[0].length() >= 11) {
234
235                                if (columnas[0].substring(0, 9).compareTo("SinNombre") == 0)
236                                        this.nombre = "SinNombre" + String.format("%02d", ++jugadoresSinNombre);
237                                else
238                                        this.nombre = columnas[0];
239
240                        } else if (columnas[0] != null)
241                                this.nombre = columnas[0];
242                        else
243                                this.nombre = "SinNombre" + String.format("%02d", ++jugadoresSinNombre);
244
245                        // Y esto viene lo ultimo porque el catch usa el nombre del jugador.
246                        try {
247                                this.apuesta = new Apuesta(columnas[7]);
248                        } catch (CantidadCamposCSVException e) {
249//                              System.err.println(e.getMessage());
250                                System.err.printf("Apuesta de %s fijada a 0.%n", this.nombre);
251                                this.apuesta = new Apuesta();
252
253                        }
254
255                } catch (NumberFormatException e) {
256                        throw new ContenidoCorruptoCSVException("No se ha podido cargar un Jugador del CSV: " + lineaDelCSV);
257                }
258
259        }

325         * Varia el crédito del jugador. Suma la cantidad pasada al credito del jugador.

329         * 
330         * @param cantidad de crédito a sumar.
331         */
332        public void variarCredito(int cantidad) {
333
334                this.credito += cantidad;
335
336                if (cantidad > 0)
337                        ganado += cantidad;
