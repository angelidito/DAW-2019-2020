004package angelidito.laruleta;
005
006import java.util.ArrayList;
007
008/**
009 * Contiene, recupera y administra a los jugadores.
010 * 
011 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
012 */
013public class ListaJugadores {
014
015        /**
016         * Lista de jugadores en activo.
017         */
018        private ArrayList<Jugador> jugadoresEnMesa;
019
020        /**
021         * Lista de jugadores retirados.
022         */
023        private static ArrayList<Jugador> jugadoresRetirados;
024
025        /**
026         * Identificador del objeto y los CSV que crea. El propósito se este parámetro
027         * es evitar que se sobre escriban los ficheros en caso de crear más de un
028         * objeto de esta clase, ya sea por error o por futuras versiones del programa.
029         */
030        private int id;
031
032        private static int numListasJugadores;
033
034        static {
035                numListasJugadores = -1;
036                jugadoresRetirados = null;
037        }
038
039        /**
040         * @return the id
041         */
042        public int getId() {
043                return id;
044        }
045
046        /**
047         * @return jugadoresEnMesa
048         */
049        public ArrayList<Jugador> getJugadoresEnMesa() {
050                return jugadoresEnMesa;
051        }
052
053        /**
054         * @return jugadoresRetirados
055         */
056        public ArrayList<Jugador> getJugadoresRetirados() {
057                return jugadoresRetirados;
058        }
059
060        /**
061         * Recupera los datos de jugadoresEnMesa y jugadoresRetirados. En caso de no
062         * existir jugadoresEnMesa.csv y jugadoresRetirados.csv, informa de no
063         * encontrarlos.
064         */
065        public ListaJugadores() {
066
067                ListaJugadores.numListasJugadores++;
068
069                this.id = ListaJugadores.numListasJugadores;
070
071                this.jugadoresEnMesa = new ArrayList<Jugador>();
072                this.cargarJugadoresEnMesa();
073
074                if (ListaJugadores.jugadoresRetirados == null) {
075                        ListaJugadores.jugadoresRetirados = new ArrayList<Jugador>();
076                        this.cargarJugadoresRetirados();
077                }
078
079        }
080
081        /**
082         * Carga los jugadores del fichero jugadoresEnMesa.csv. Los guarda en los
083         * jugadoresEnMesa. En caso de no existir jugadoresEnMesa.csv, informa de no
084         * encontrarlo.
085         */
086        private void cargarJugadoresEnMesa() {
087
088                String nombreFichero = String.format("jugadoresEnMesa%d.csv", id);
089
090                this.jugadoresEnMesa = GestionCSV.obtenerJugadores(nombreFichero);
091
092        }
093
094        /**
095         * Carga los jugadores del fichero jugadoresEnMesa.csv. Los guarda en los
096         * jugadoresEnMesa. En caso de no existir y jugadoresRetirados.csv, informa de
097         * no encontrarlo.
098         */
099        private void cargarJugadoresRetirados() {
100
101                String nombreFichero = "jugadoresRetirados.csv";
102
103                ListaJugadores.jugadoresRetirados = GestionCSV.obtenerJugadores(nombreFichero);
104
105        }
106
107        /**
108         * Guarda los jugadores en los CSV. Para cada ArrayList guarda los jugadores en
109         * un CSV.
110         */
111        public void guardarJugadores() {
112
113                String ficheroEnMesa = String.format("jugadoresEnMesa%d.csv", id);
114                String ficheroRetirados = "jugadoresRetirados.csv";
115
116                GestionCSV.escribirCSV(ficheroEnMesa, jugadoresEnMesa);
117                GestionCSV.escribirCSV(ficheroRetirados, jugadoresRetirados);
118
119        }
120
121        /**
122         * Comprueba si hay que retirar al jugador. Esto es cuando no le queda crédito,
123         * ha alcanzado el credito objetivo o ha alcanzado las rondas máximas.
124         * 
125         * @param jugadorParaComprobar jugador qu ehay que comprobar la retirada
126         * @return {@code true} si hay que retirarlo; {@code false} si no.
127         */
128        public boolean comprobarRetidadaJugador(Jugador jugadorParaComprobar) {
129                boolean hayQue = false;
130
131                if (jugadorParaComprobar.getCredito() < 1
132                                || jugadorParaComprobar.getCreditoObjetivo() <= jugadorParaComprobar.getCredito()
133                                || jugadorParaComprobar.getRondas() >= jugadorParaComprobar.getRondasMaximas())
134                        hayQue = true;
135
136                return hayQue;
137        }
138
139        /**
140         * Retira a un jugador. Lo quita de jugadores en mesa y lo pasa a jugadores
141         * retirados.
142         * 
143         * @param jugadorParaRetirar Jugador a retirar.
144         * @return Mensaje de retirada o no del jugador.
145         */
146        public String retirarJugador(Jugador jugadorParaRetirar) {
147
148                boolean retirado = false;
149
150                for (int i = 0; !retirado && i < this.jugadoresEnMesa.size(); i++) {
151                        if (this.jugadoresEnMesa.get(i).getId() == jugadorParaRetirar.getId()) {
152
153                                ListaJugadores.jugadoresRetirados.add(jugadorParaRetirar);
154
155                                this.jugadoresEnMesa.remove(i);
156                                i--;
157                                retirado = true;
158
159                        }
160                }
161
162                String mensajeRetirada = "";
163                if (retirado) {
164                        mensajeRetirada = "%s ha sido retirado.%s";
165                        if (jugadorParaRetirar.getCredito() < 0)
166                                EasterEgg.huevoDePascua(jugadorParaRetirar.getNombre());
167
168                        else {
169
170                                if (jugadorParaRetirar.getCredito() == 0)
171                                        mensajeRetirada = String.format(mensajeRetirada, jugadorParaRetirar.getNombre(),
172                                                        " Se le ha agotado el crédito.");
173
174                                else if (jugadorParaRetirar.getCredito() >= jugadorParaRetirar.getCreditoObjetivo())
175                                        mensajeRetirada = String.format(mensajeRetirada, jugadorParaRetirar.getNombre(),
176                                                        " Ha alcanzado su crédito objetivo");
177
178                                else if (jugadorParaRetirar.getRondas() >= jugadorParaRetirar.getRondasMaximas())
179                                        mensajeRetirada = String.format(mensajeRetirada, jugadorParaRetirar.getNombre(),
180                                                        " Ha alcanzado sus rondas máximas");
181
182                                else
183                                        mensajeRetirada = String.format(mensajeRetirada, jugadorParaRetirar.getNombre(), "");
184
185                        }
186
187                } else {
188                        mensajeRetirada = "No se ha podido retirar al jugador.";
189                }
190
191                return mensajeRetirada;
192
193        }
194
195        /**
196         * Devuelve un vector con la lista de jugadores en mesa
197         * 
198         * @return Vector con la lista numerada de jugadores.
199         */
200        public String[] listadoJugadoresEnMesa() {
201
202                String[] listado = new String[this.jugadoresEnMesa.size()];
203
204                for (int i = 0; i < this.jugadoresEnMesa.size(); i++)
205                        listado[i] = String.format("%d - %s", (i + 1), this.jugadoresEnMesa.get(i).info());
206
207                return listado;
208        }
209
210        /**
211         * Borra el contenido de las listas de jugadores.
212         */
213        public void borrarListas() {
214
215                this.jugadoresEnMesa = new ArrayList<Jugador>();
216                ListaJugadores.jugadoresRetirados = new ArrayList<Jugador>();
217
218        }
