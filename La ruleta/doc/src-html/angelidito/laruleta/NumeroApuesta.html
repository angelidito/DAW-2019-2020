004package angelidito.laruleta;

006/**
007 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
008 *
009 */
010public class NumeroApuesta extends NumeroTablero {
011        private static final long serialVersionUID = -3117945003343960634L;
012
013        /**
014         * Cantidad apostada al número {@code n}.
015         */
016        private int cantidadApostada = 0;
017
018        /**
019         * @return la cantidad apostada.
020         */
021        public int getCantidadApostada() {
022                return cantidadApostada;
023        }
024
025        /**
026         * @param cantidad La cantidad a apostar.
027         */
028        public void setCantidadApostada(int cantidad) {
029                this.cantidadApostada = cantidad;
030        }
031
032        /**
033         * No apuesta nada. A los demás atributos de la clase los encaja en los valores
034         * que les corresponden en un tablero de ruleta francesa. Si el número no existe
035         * en el tablero o es 0, los valores de todos los atributos salvo de {@code n} y
036         * {@code cantidadApostada} seran iguales -1.
037         * 
038         * 
039         * @param numero El numero al que corresponde el objeto.
040         */
041        public NumeroApuesta(int numero) {
042                super(numero);
043                this.cantidadApostada = 0;
044        }
045
046        /**
047         * Apuesta la cantidad pasada al número {@code n}. A los demás atributos de la
048         * clase los encaja en los valores que les corresponden en un tablero de ruleta
049         * francesa. Si el número no existe en el tablero o es 0, los valores de todos
050         * los atributos salvo de {@code n} y {@code cantidadApostada} seran iguales -1.
051         * 
052         * @param numero   El numero al que corresponde el objeto.
053         * @param cantidad Cantidad a apostar.
054         */
055        public NumeroApuesta(int numero, int cantidad) {
056                super(numero);
057                this.cantidadApostada = cantidad;
058        }
059
060        @Override
061        public String toString() {
062                return String.format("%d al nº%d", cantidadApostada, getN());
063        }
064
065}
