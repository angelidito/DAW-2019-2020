001package angelidito.laruleta;
002
003public class NumeroRuleta extends NumeroTablero {
004        private static final long serialVersionUID = -6284727951773680634L;
005
006        /**
007         * Ocurrencias del número {@code n}.
008         */
009        private int ocurrencias = 0;
010
011        /**
012         * @return Las ocurrencias del número.
013         */
014        public int getOcurrencias() {
015                return ocurrencias;
016        }
017
018        /**
019         * @param ocurrencias Las ocurrencias a guardar.
020         */
021        public void setOcurrencias(int ocurrencias) {
022                this.ocurrencias = ocurrencias;
023        }
024
025        /**
026         * Añadie una ocurrencia al número.
027         */
028        public void añadirOcurrencia() {
029                ocurrencias++;
030
031        }
032
033        public NumeroRuleta(int numero) {
034                super(numero);
035                this.ocurrencias = 0;
036        }
037
038        public NumeroRuleta(int numero, int ocurrencias) {
039                super(numero);
040                this.ocurrencias = ocurrencias;
041
042        }
043
044        @Override
045        public String toString() {
046                String str = String.format("Nº%02d - %d ocurrencias", this.getN(), this.ocurrencias);
047                return str;
048        }
049
050}
