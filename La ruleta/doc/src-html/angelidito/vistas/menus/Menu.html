004package angelidito.vistas.menus;

006import angelidito.vistas.Vista;
007
008/**
009 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
010 *
011 */
012public class Menu extends Vista {
013
014        private String cabecera;
015        private String[] opciones;
016
017        /**
018         * Crea la vista.
019         * 
020         * @param cabecera Cabecera del menú.
021         * @param opciones Vector con las opciones imprimir.
022         */
023        public Menu(String cabecera, String[] opciones) {
024
025                this.cabecera = cabecera;
026                this.opciones = opciones;
027
028        }
029
030        /**
031         * @param cabecera the cabecera to set
032         */
033        protected void setCabecera(String cabecera) {
034                this.cabecera = cabecera;
035        }
036
037        /**
038         * @param opciones the opciones to set
039         */
040        protected void setOpciones(String[] opciones) {
041                this.opciones = opciones;
042        }
043
044        /**
045         * @return the opciones
046         */
047        protected String[] getOpciones() {
048                return opciones;
049        }
050
051        /**
052         * Imprime el menú. Cabecera y opciones.
053         */
054        public void printMenu() {
055                System.out.println(cabecera);
056                for (String opcion : opciones) {
057                        System.out.println(opcion.toString());
058                }
059                println();
060
061        }
062
063        /**
064         * Imprime el menú. Cabecera y opciones.
065         * 
066         * @param opcionIncorrecta Si es positivo imprime la un mensaje.
067         */
068        public void printMenu(boolean opcionIncorrecta) {
069
070                System.out.println(cabecera);
071
072                for (String op : opciones)
073                        System.out.println(op);
074
075//              for (int i = 0; i < opciones.length; i++) 
076//                      System.out.println(opciones[i]);
077
078                if (opcionIncorrecta)
079                        System.out.println("Asegúrese de que introduce una opción de entre las disponibles.");
080                else
081                        println();
082
083        }
084
085}
