004package angelidito.vistas.menus;
005
006import angelidito.laruleta.Jugador;
007import angelidito.laruleta.ListaJugadores;
008import angelidito.vistas.Escaner;
009
010/**
011 * Menú con las opciones para editar a un jugador. Permite la edición de los
012 * jugadores en mesa. Permite editar nombres, crédito, crédito objetivo, rondas
013 * máximas y apuestas.
014 * 
015 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
016 *
017 */
018public class MenuEditarJugadores extends Menu {
019
020        private final static String CABECERA = "Retirada de jugadores\n  Seleccione un jugador:";
021
022        /**
023         * Lista de jugadores que editar.
024         */
025        private ListaJugadores listaJugadores;
026
027        /**
028         * Crea la vista.
029         * 
030         * @param listaJugadores Lista de jugadores que editar.
031         */
032        public MenuEditarJugadores(ListaJugadores listaJugadores) {
033                this(CABECERA, listaJugadores.listadoJugadoresEnMesa(), listaJugadores);
034
035        }
036
037        /**
038         * Crea la vista.
039         * 
040         * @param cabecera       Cabecera del menú.
041         * @param opciones       Vector con las opciones del menú.
042         * @param listaJugadores Lista de jugadores que editar.
043         */
044        private MenuEditarJugadores(String cabecera, String[] opciones, ListaJugadores listaJugadores) {
045                super(cabecera, opciones);
046                this.listaJugadores = listaJugadores;
047        }
048
049        /**
050         * Permite seleccionar y editar a un jugador.
051         */
052        public void editarJugador() {
053                int opcion;
054                boolean retirar;
055                Jugador jugadorParaEditar = selecionarJugador();
056
057                do {
058
059                        System.out.printf("Ha slecionado a %s%n%n", jugadorParaEditar.info());
060
061                        System.out.println("  Escoja una opción:");
062                        System.out.println("1 - Cambiar nombre");
063                        System.out.println("2 - Variar crédito");
064                        System.out.println("3 - Cambiar crédito objetivo");
065                        System.out.println("4 - Cambiar rondas máximas");
066                        System.out.println("5 - Cambiar apuesta");
067                        System.out.println("0 - Guardar cambios");
068                        System.out.println("");
069
070                        opcion = Escaner.entero(0, 5);
071
072                        System.out.println();
073
074                        retirar = editarJugador(opcion, jugadorParaEditar);
075
076                        System.out.println();
077
078                } while (opcion != 0);
079
080                if (retirar) {
081                        String mensajeRetirada;
082                        mensajeRetirada = listaJugadores.retirarJugador(jugadorParaEditar);
083                        System.out.println(mensajeRetirada);
084                }
085
086        }
087
088        /**
089         * Edita el jugador según la opcion introducida por teclado. Opciones: 1,
090         * nombre; 2, crédito; 3, crédito objetivo;, 4, rondas máximas; 5, apuesta; 0,
091         * nada.
092         * 
093         * @param opcion            1, para cambiar el nombre; 2, para variar el
094         *                          crédito; 3, para cambiar el crédito objetivo; 4,
095         *                          para cambiar el máximo número de rondas; 5, para
096         *                          editar la apuesta; 0, no hace nada; en cualquier
097         *                          otro caso, informa que la opción es incorrecta.
098         * @param jugadorParaEditar Jugador a editar.
099         * @return {@code true} si el jugador ha alcanzado el crédito objetivo o las
100         *         rondas máximas, o si no tiene crédito; de otra manera, {@code false}
101         */
102        private boolean editarJugador(int opcion, Jugador jugadorParaEditar) {
103
104                boolean retirar = false;
105
106                switch (opcion) {
107
108                case 1:
109                        System.out.println("Introduzca el nuevo nombre.");
110
111                        jugadorParaEditar.setNombre(Escaner.texto());
112
113                        break;
114
115                case 2:
116                        System.out.println("Inserte la cantidad a sumar (restará si es negativa).");
117
118                        jugadorParaEditar.variarCredito(Escaner.entero(-1 * Jugador.MAX_INT_VALUE + jugadorParaEditar.getCredito(),
119                                        Jugador.MAX_INT_VALUE - jugadorParaEditar.getCredito()));
120
121                        if (jugadorParaEditar.getCredito() < 1) {
122                                retirar = true;
123                                System.err.println("Sin credito. El jugador se retirará tras la edición");
124                        }
125
126                        break;
127
128                case 3:
129                        System.out.println("Inserte el nuevo crédito objetivo.");
130                        System.out.printf("El actual es: %d.%n", jugadorParaEditar.getCreditoObjetivo());
131
132                        int nuevoCreditoObjetivo = Escaner.entero(0, Jugador.MAX_INT_VALUE);
133                        jugadorParaEditar.setCreditoObjetivo(nuevoCreditoObjetivo);
134
135                        if (jugadorParaEditar.getCreditoObjetivo() <= jugadorParaEditar.getCredito()) {
136                                retirar = true;
137                                System.err.println("Crédito objetivo superado. El jugador se retirará tras la edición");
138                        }
139
140                        break;
141
142                case 4:
143                        System.out.println("Inserte el nuevo máximo de rondas.");
144                        System.out.printf("El actual es: %d.%n", jugadorParaEditar.getRondasMaximas());
145
146                        int nuevasRondasMaximas = Escaner.entero(1, Jugador.MAX_INT_VALUE);
147                        jugadorParaEditar.setCreditoObjetivo(nuevasRondasMaximas);
148
149                        if (jugadorParaEditar.getRondasMaximas() <= jugadorParaEditar.getRondas()) {
150                                retirar = true;
151                                System.err.println("Rondas máximas superadas. El jugador se retirará tras la edición");
152                        }
153
154                        break;
155
156                case 5:
157
158                        int op;
159                        boolean opcionIncorrecta = false;
160                        int[] opciones = MenuEditarApuesta.getNumOpciones();
161                        do {
162
163                                MenuEditarApuesta edicionApuesta = new MenuEditarApuesta(jugadorParaEditar);
164
165                                edicionApuesta.printMenu(opcionIncorrecta);
166
167                                op = Escaner.entero();
168
169                                opcionIncorrecta = true;
170
171                                for (int i = 0; opcionIncorrecta && i < opciones.length; i++) {
172                                        if (opciones[i] == op)
173                                                opcionIncorrecta = false;
174                                }
175
176                                if (!opcionIncorrecta)
177                                        edicionApuesta.editarApuesta(op);
178
179                        } while (op != 0);
180
181                        break;
182
183                case 0:
184                        // NO TIENE QUE HACER NADA.
185                        // Solo salir.
186                        break;
187
188                }
189                return retirar;
190        }
191
192        /**
193         * Pide selecionar por teclado un jugador de la lista de jugadores en mesa.
194         * 
195         * @return El Jugador selecionado.
196         */
197        private Jugador selecionarJugador() {
198
199                Jugador jugadorSelecionado = null;
200
201                do {
202
203                        int numeroJugador = Escaner.entero(1, super.getOpciones().length) - 1;
204
205                        jugadorSelecionado = listaJugadores.getJugadoresEnMesa().get(numeroJugador);
206
207                } while (jugadorSelecionado == null);
208
209                return jugadorSelecionado;
210        }
