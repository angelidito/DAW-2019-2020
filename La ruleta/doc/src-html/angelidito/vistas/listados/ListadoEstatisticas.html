006/**
007 * Listado de las estadisticas de la ruleta.
008 * 
009 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
010 *
011 */
012public class ListadoEstatisticas extends Listado {
013
014        /**
015         * Metodo unico de la clase. Crea la vista y muestra las estadisticas de la
016         * rulta y las canancias del casino.
017         * 
018         * @param estadisticas Estadísticas de la ruleta.
019         * @param ganancias    Ganancias del día del casino.
020         */
021        public ListadoEstatisticas(String estadisticas, int ganancias) {
022                System.out.println(estadisticas);
023
024                System.out.println("");
025
026                if (1 < ganancias)
027                        System.out.printf("Hoy la casa ha ganado %d créditos.%n", ganancias);
028                else if (ganancias < -1)
029                        System.out.printf("Hoy la casa ha perdido %d créditos.%n", -1 * ganancias);
030                else if (ganancias == 1)
031                        System.out.printf("Hoy la casa ha ganado 1 crédito.%n");
032                else if (ganancias == 0)
033                        System.out.printf("Hoy la casa no ha ganado nada.%n");
034                else if (ganancias == -1)
035                        System.out.printf("Hoy la casa ha perdido 1 crédito.%n");
036
037                System.out.println("");
038
039                waitForEnter();
040
041                System.out.println("");
042
043        }
044
045}
