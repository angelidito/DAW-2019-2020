061         * Escanea un número entero positivo por teclado. El método guía al usuario si
062         * se empeña en no intruducir un número analizable hasta que lo hace.
063         * 
064         * @return El número dentro del rango introducido por el usuario.
065         */

067                checkNotClosed();
068
069                String textoEscaneado;
070                int numeroIntroducido = 0;
071                boolean textoEsNumero = false;
072
073                do {
074
075                        textoEscaneado = escaner.nextLine();
076                        System.out.println();
077
078                        try {
079                                numeroIntroducido = Integer.parseInt(textoEscaneado);
080                                textoEsNumero = true;
081
082                        } catch (NumberFormatException e) {
083
084                                System.out.println("Error. Introduzca un número.");
085                                textoEsNumero = false;
086
087                        }
088
089                } while (!textoEsNumero);
090
091                // escaner.close();
092
093                return numeroIntroducido;
094

136        /**
137         * Escanea el número entero introducido por teclado. Sólo aceptará un número
138         * entero que se encuentre entre los parámetros min y max, ambos inclusive. El
139         * método guía al usuario si se empeña en no intruducir un número analizable
140         * hasta que lo hace. Si min es mayor que max o max es mayor que min, ambos se
141         * fijaran en el mayor de los dos.
142         * 
143         * @param min Valor mínimo der rango
144         * @param max Valor máximo der rango
145         * @return El número escaneado dentro del rango.
146         */
147        public static int entero(int min, int max) {
148                checkNotClosed();
149
150                if (min > max || max < min) {
151                        min = Math.max(max, min);
152                        max = min;
153                }
154
155                String textoEscaneado = "0x80000000";
156                int numeroIntroducido = -999999;
157                boolean textoEsNumero = false;
158
159                do {
160
161                        textoEscaneado = escaner.nextLine();
162                        System.out.println();
163
164                        try {
165                                numeroIntroducido = Integer.parseInt(textoEscaneado);
166                                textoEsNumero = true;
167
168                                if (!(min <= numeroIntroducido && numeroIntroducido <= max))
169                                        System.out.printf("El número debe estar entre %d y %d, ambos inclusive.%n", min, max);
170
171                        } catch (NumberFormatException e) {
172
173                                System.out.println("Error. Introduzca de nuevo el número.");
174                                textoEsNumero = false;
175
176                        }
177
178                } while (!textoEsNumero || !(min <= numeroIntroducido && numeroIntroducido <= max));
179
180                // escaner.close();
181                return numeroIntroducido;
182
183        }

185        /**
186         * Escanea el dígito introducido por teclado y lo devuelve como un carácter.
187         * Sólo aceptará un dígito que se encuentre entre los parámetros min y max,
188         * ambos inclusive. El método guía al usuario si se empeña en no intruducir lo
189         * que debe hasta que lo hace. Tanto el mínimo como el máximo deben estar entre
190         * 0 y 9, ambos inclusive. Si min es mayor que max o max es mayor que min, ambos
191         * se fijaran en el mayor de los dos.
192         * 
193         * @param min Mínimo valor del dígito
194         * @param max Máximo valor del dijito
195         * @return El carácter correspondiente al número dentro del rango introducido
196         *         por el usuario.
197         */
198        public static char caracterNumericoEnRango(int min, int max) {
199                checkNotClosed();
200                if (min < 0)
201                        min = 0;
202                if (max > 9)
203                        max = 9;
204                if (min > max || max < min) {
205                        min = Math.max(max, min);
206                        max = min;
207                }
208
209                String textoEscaneado;
210
211                textoEscaneado = escaner.nextLine();
212                System.out.println();
213
214                while (textoEscaneado.length() != 1 || !Character.isDigit(textoEscaneado.charAt(0))
215                                || !(min <= Integer.parseInt(textoEscaneado.substring(0, 1))
216                                                && Integer.parseInt(textoEscaneado.substring(0, 1)) <= max)) {
217
218                        if (textoEscaneado.length() != 1) {
219                                System.out.println("Debe introducir un único número.");
220                        } else {
221                                System.out.println("Por favor, escriba un solo número.");
222                                System.out.printf("Debe de estar entre %d y %d.%n", min, max);
223                        }
224
225                        textoEscaneado = escaner.nextLine();
226                        System.out.println();
227
228                }
229
230                return textoEscaneado.charAt(0);
231        }

269        /**
270         * Pide un si o un no y devuelve en consecuencia {@code true} o {@code false}.
271         * Escanea el texto introducido por teclado hasta que se introduzca una de las
272         * siguiente opciones: s, si, sí, n, no. No atiende a mayúsculas o minúsculas.
273         * 
274         * @return {@code true} o {@code false}, dependiendo si la respusta es si o no.
275         */
276        public static boolean yesNoQuestion() {
277                checkNotClosed();
278
279                String texto = "";
280                boolean textoAdecuado = false;
281                boolean yesNo = false;
282                do {
283                        try {
284                                texto = escaner.nextLine();
285                        } catch (Exception e) {
286                                texto = "";
287                        }
288                        if (texto.compareToIgnoreCase("s") == 0 || texto.compareToIgnoreCase("sí") == 0
289                                        || texto.compareToIgnoreCase("si") == 0) {
290
291                                textoAdecuado = true;
292                                yesNo = true;
293
294                        } else if (texto.compareToIgnoreCase("n") == 0 || texto.compareToIgnoreCase("no") == 0) {
295                                textoAdecuado = true;
296                                yesNo = false;
297
298                        }
299
300                } while (!textoAdecuado);
301
302                return yesNo;
303        }

393        /**
394         * Abre de nuevo el escaner. Si y sólo si estába cerrado. Si no, no hace nada.
395         */
396        public void open() {
397                if (closed)
398                        escaner = new Scanner(System.in);
399        }
400
401        /**
402         * Si el escaner no está cerrado, lo cierra.
403         */
404        @Override
405        public void close() {
406                if (!closed) {
407                        escaner.close();
408                        closed = true;
409                }
410
411        }

413        /**
414         * Lanza ClosedEscanerRTException si el escaner está cerrado. Si está abierto no
415         * hace nada.
416         * 
417         * @throws ClosedEscanerRTException Si el escaner está cerrado.
418         */
419        private static void checkNotClosed() {
420                if (closed)
421                        throw new ClosedEscanerRTException("El escaner está cerrado.");
422        }
