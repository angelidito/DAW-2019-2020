004package angelidito.vistas;

006/**
007 * 
008 * Clase madre de las vistas. Inlcluye tres métodos: println, waitForEnter y
009 * mostrar.
010 * 
011 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
012 *
013 */
014public class Vista {
015
016        /**
017         * Imprime por pantalla un salto de linea. Lo sé, muy inútil, era para ahorrarme
018         * algo de tiempo escribiendo saltos de linea.
019         */
020        public void println() {
021                System.out.println();
022        }
023
024        /**
025         * Espera a que el usuario pulse enter antes de continuar.
026         */
027        public void waitForEnter() {
028                System.out.println("Pulsa Enter para continuar...");
029                Escaner.Enter();
030        }
031
032        /**
033         * Realiza un System.out.println del parámetro.
034         * 
035         * @param <O> Tipo del objeto a mostrar.
036         * 
037         * @param str El objeto a imprimir por pantalla.
038         */
039        public <O> void mostrar(O str) {
040                System.out.println(str);
041        }
042}
