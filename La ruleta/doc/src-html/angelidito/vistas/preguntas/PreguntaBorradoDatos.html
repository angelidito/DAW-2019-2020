008/**
009 * Pregunta sobre el borrado de datos.
010 * 
011 * @author <a href="https://twitter.com/angelidito">Ángel M. D.</a>
012 *
013 */
014public class PreguntaBorradoDatos extends Preguntas {
015
016        /**
017         * 
018         * Pregunta si de desean borrar los datos. Imprime por pantalla si la acción se
019         * va o no a llevar a cabo.
020         * 
021         * @return {@code true} si e quieren borrar, {@code false} si no.
022         */
023        public boolean borrarDatos() {
024
025                System.out.println("Está a punto de borrar los archivos del programa.");
026                System.out.print("Esta acción no puede deshacerse. ¿Esta seguro?");
027
028                boolean borrar = Escaner.yesNoQuestionRecursivo();
029
030                datosBorrados(borrar);
031
032                return borrar;
033        }
034
035        /**
036         * Imprime por pantalla si el borrado se va a llevar a cabo o no. Dependiendo de
037         * si el parámetro es true o false.
038         * 
039         * @param borrado Confirmación o no del borrado.
040         */
041        private void datosBorrados(boolean borrado) {
042
043                if (borrado)
044                        System.out.println("Borrando datos.");
045
046                else
047                        System.out.println("No re realiza ningun borrado.");
048
049                println();
050
051        }
052}
